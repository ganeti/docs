31  """Module holding different constants."""

33  # pylint: disable=W0401,W0614 
34  # 
35  # The modules 'ganeti._constants' and 'ganeti._vcsversion' are meant 
36  # to be re-exported but pylint complains because the imported names 
37  # are not actually used in this module. 
38   
39  import re 
40  import socket 
41   
42  from ganeti._constants import * 
43  from ganeti._vcsversion import * 
44  from ganeti import compat 
45  from ganeti import pathutils 
46   
47  ALLOCATABLE_KEY = "allocatable" 
48  FAILED_KEY = "failed" 
49   
50  DAEMONS_LOGFILES = \ 
51      dict((daemon, pathutils.GetLogFilename(DAEMONS_LOGBASE[daemon])) 
52           for daemon in DAEMONS_LOGBASE) 
53   
54  DAEMONS_EXTRA_LOGFILES = \ 
55    dict((daemon, dict((extra, 
56         pathutils.GetLogFilename(DAEMONS_EXTRA_LOGBASE[daemon][extra])) 
57         for extra in DAEMONS_EXTRA_LOGBASE[daemon])) 
58           for daemon in DAEMONS_EXTRA_LOGBASE) 
59   
60  IE_MAGIC_RE = re.compile(r"^[-_.a-zA-Z0-9]{5,100}$") 
61   
62  # External script validation mask 
63  EXT_PLUGIN_MASK = re.compile("^[a-zA-Z0-9_-]+$") 
64   
65  JOB_ID_TEMPLATE = r"\d+" 
66  JOB_FILE_RE = re.compile(r"^job-(%s)$" % JOB_ID_TEMPLATE) 
67   
68  # HVC_DEFAULTS contains one value 'HV_VNC_PASSWORD_FILE' which is not 
69  # a constant because it depends on an environment variable that is 
70  # used for VClusters.  Therefore, it cannot be automatically generated 
71  # by Haskell at compilation time (given that this environment variable 
72  # might be different at runtime). 
73  HVC_DEFAULTS[HT_XEN_HVM][HV_VNC_PASSWORD_FILE] = pathutils.VNC_PASSWORD_FILE 
74   
75  # Do not re-export imported modules 
76  del re, socket, pathutils, compat
