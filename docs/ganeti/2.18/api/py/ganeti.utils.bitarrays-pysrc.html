52    @param reserve: whether to reserve the first available slot or not 
53    @return: the idx of the (first) available slot 
54    @raise errors.OpPrereqError: If all slots in a bitarray are occupied 
55      or the given slot is not free. 
56   
57    """ 
58    if slot is not None: 
59      assert slot < len(slots) 
60      if slots[slot]: 
61        raise errors.GenericError("Slot %d occupied" % slot) 
62   
63    else: 
64      avail = slots.search(_AVAILABLE_SLOT, 1) 
65      if not avail: 
66        raise errors.GenericError("All slots occupied") 
67   
68      slot = int(avail[0]) 
69   
70    if reserve: 
71      slots[slot] = True 
72
